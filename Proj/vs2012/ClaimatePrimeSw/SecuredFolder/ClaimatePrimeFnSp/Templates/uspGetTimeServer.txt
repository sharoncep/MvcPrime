CREATE PROCEDURE [dbo].[usp_GetTime_Server]
AS
BEGIN
	SET NOCOUNT ON;

    DECLARE @TBL_ANS TABLE
    (
		[ID] INT NOT NULL IDENTITY (1, 1)
		, [CURR_DT_TM] DATETIME NOT NULL
    );
    
    INSERT INTO @TBL_ANS VALUES (GETDATE());
		
	SELECT * FROM @TBL_ANS;
	
	-- EXEC [dbo].[usp_GetTime_Server]
END

GO
